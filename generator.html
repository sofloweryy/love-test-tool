    <script>

        // 数据存储核心函数

        const initData = () => {

            const data = localStorage.getItem('testLinkData');

            if (!data) {

                localStorage.setItem('testLinkData', JSON.stringify({

                    basic: { total: 0, used: 0, records: [] },

                    pro: { total: 0, used: 0, records: [] }

                }));

            }

        };


        const getData = () => {

            initData();

            return JSON.parse(localStorage.getItem('testLinkData'));

        };


        const saveData = (data) => {

            localStorage.setItem('testLinkData', JSON.stringify(data));

        };


        const addLinkRecord = (version, uuid, createTime) => {

            const data = getData();

            data[version].total += 1;

            data[version].records.push({

                uuid,

                createTime,

                status: 'unused',

                useTime: ''

            });

            saveData(data);

            updateStockDisplay();

            updateRecordDisplay();

        };


        const updateLinkStatus = (version, uuid, status) => {

            const data = getData();

            const record = data[version].records.find(item => item.uuid === uuid);

            if (record) {

                record.status = status;

                if (status === 'used') {

                    record.useTime = new Date().toString();

                    data[version].used += 1;

                }

                saveData(data);

                updateStockDisplay();

                updateRecordDisplay();

            }

        };


        const updateStockDisplay = () => {

            const data = getData();

            document.getElementById('basic-total').textContent = data.basic.total;

            document.getElementById('basic-used').textContent = data.basic.used;

            document.getElementById('basic-left').textContent = data.basic.total - data.basic.used;

            document.getElementById('pro-total').textContent = data.pro.total;

            document.getElementById('pro-used').textContent = data.pro.used;

            document.getElementById('pro-left').textContent = data.pro.total - data.pro.used;

        };


        const updateRecordDisplay = (searchKey = '') => {

            const data = getData();

            const basicRecords = data.basic.records

                .filter(item => searchKey === '' || item.uuid.includes(searchKey))

                .map(item => `

                    <div class="record-item">

                        <div class="uuid">${item.uuid}</div>

                        <div>

                            <span class="status status-${item.status}">${

                                item.status === 'unused' ? '未使用' : 

                                item.status === 'used' ? '已核销' : '已过期'

                            }</span>

                            ${item.useTime ? `<br/>核销时间：${item.useTime.slice(0, 24)}` : ''}

                        </div>

                    </div>

                `).join('') || '<div style="text-align:center; color:#666; padding:10px;">暂无记录</div>';

            document.getElementById('basicRecords').innerHTML = basicRecords;


            const proRecords = data.pro.records

                .filter(item => searchKey === '' || item.uuid.includes(searchKey))

                .map(item => `

                    <div class="record-item">

                        <div class="uuid">${item.uuid}</div>

                        <div>

                            <span class="status status-${item.status}">${

                                item.status === 'unused' ? '未使用' : 

                                item.status === 'used' ? '已核销' : '已过期'

                            }</span>

                            ${item.useTime ? `<br/>核销时间：${item.useTime.slice(0, 24)}` : ''}

                        </div>

                    </div>

                `).join('') || '<div style="text-align:center; color:#666; padding:10px;">暂无记录</div>';

            document.getElementById('proRecords').innerHTML = proRecords;

        };

        // 生成UUID

        const generateUUID = () => {

            return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {

                const r = Math.random() * 16 | 0;

                const v = c === 'x' ? r : (r & 0x3 | 0x8);

                return v.toString(16);

            });

        };

            }
